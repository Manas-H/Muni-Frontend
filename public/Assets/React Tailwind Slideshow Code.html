  3. Create the Slideshow component: In your src folder, create a new file called Slideshow.js and add the following code:

    jsx
    import React, { useState } from 'react';
import './Slideshow.css'; // Import the custom CSS for the active dot class

const slides = [
  {
    author: 'Employers',
    quote: 'Streamline your recruitment process with AI-driven tools for efficient hiring.'
  },
  {
    author: 'Students',
    quote: 'Unlock your potential with personalised job matches and seamless application tracking.'
  },
  {
    author: 'Universities',
    quote: 'Connect your students with top employers and track engagement effortlessly.'
  }
];

const Slideshow = () => {
  const [currentSlide, setCurrentSlide] = useState(2);

  const nextSlide = () => {
    setCurrentSlide((prevSlide) => (prevSlide === slides.length - 1 ? 0 : prevSlide + 1));
  };

  const prevSlide = () => {
    setCurrentSlide((prevSlide) => (prevSlide === 0 ? slides.length - 1 : prevSlide - 1));
  };

  const setSlide = (index) => {
    setCurrentSlide(index);
  };

  return (
    <div className="container mx-auto">
      <div className="relative overflow-hidden w-full h-full">
        {slides.map((slide, index) => (
          <div
            key={index}
            className={`mySlides ${index === currentSlide ? 'block' : 'hidden'}`}
          >
            <p className="author text-xl font-bold">{slide.author}</p>
            <q className="text-lg">{slide.quote}</q>
          </div>
        ))}
        <button className="prev absolute left-0 top-1/2 transform -translate-y-1/2" onClick={prevSlide}></button>
        <button className="next absolute right-0 top-1/2 transform -translate-y-1/2" onClick={nextSlide}></button>
      </div>
      <div className="dot-container flex justify-center mt-4">
        {slides.map((_, index) => (
          <span
            key={index}
            className={`dot w-4 h-4 mx-1 rounded-full cursor-pointer ${index === currentSlide ? 'bg-gray-800' : 'bg-gray-400'}`}
            onClick={() => setSlide(index)}
          ></span>
        ))}
      </div>
    </div>
  );
};
